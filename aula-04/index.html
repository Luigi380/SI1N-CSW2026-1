def equacao_segundo_grau(a, b, c):
    if a == 0:
        return "Não é uma equação de segundo grau."

    delta = b**2 - 4*a*c

    if delta < 0:
        return "A equação não possui raízes reais."
    
    x1 = (-b + math.sqrt(delta)) / (2*a)
    x2 = (-b - math.sqrt(delta)) / (2*a)

    return x1, x2
